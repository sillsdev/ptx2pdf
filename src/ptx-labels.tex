\def\labelp@ge#1{%
    \write\picp@ges{\string\labelonpage{\folio}#1}%
}
\x@\def\csname start-zlabel\endcsname{%
  \get@ttribute{id}%
  \trace{m}{zlabel: \attr@b}%
  \ifx\attr@b\relax
    \let\attr@b\c@rref
  \fi
  \ifx\attr@b\relax\else
    \edef\l@gstring{{\attr@b}{\c@rref}}%
    \trace{m}{zlabel: \l@gstring}%
    \x@\labelp@ge\x@{\l@gstring}%
  \fi
}
\x@\def\csname start-zpage\endcsname{%
  \get@ttribute{label}%
  \ifx\attr@b\relax\else
    \csname zpage-\attr@b\endcsname
  \fi
}
  
\def\labelonpage#1#2#3{%
  \x@\def\csname zpage-#2\endcsname{#1}%Page number for label.
  \x@\def\csname zref-#2\endcsname{#3}%Reference for label. (why??)
}
