# strip out \ms1 everywhere - I need to get rid of this when I have permanently removed \ms1 from projects
# '\\ms1 .+?\n'>''

# REMOVE entire imte1 line we only remove them in monoglots
# But in 2024-08 they are such a pain in the neck I'm taking them out of diglots too. 
'\\imte1 .*?\r?\n'>''

# REMOVE entire \rem line 
'\\rem .*?\r?\n'>''

# REMOVE entire \ziex line 
'\\ziex .*?\r?\n'>''

# REPLACE \li3 with \li to make lists of traits appear as normal lists
# We normally leave the # sign out of this command
# even if we want to print normal paragraphs because the next command controls that
'\\li3'>'\\li '

# REMOVE \li markers to print lists marked with \li as normal paragraphs
# Leave the # symbol in place to print \li paragraphs as lists, possibly for final checking 
# REMOVE the # symbol to print normal paragraphs. 
'\\li '>''
# Don't print chapter numbers in TDX
at TDX "\\c \d+" > ""
# Don't print verse numbers in TDX
at TDX "\\v \d+" > ""
# Format TDX as single column on section headings and double column on reference lists.
at TDX "(\\zs[123].+?\n)" > "\1\n\\doublecolumns\n"
at TDX "(\\s[123].+?\n)" > "\\p\n\\singlecolumn\n\1"

# Don't print verse numbers in Introduction
at INT "\\v \d+ " > ""

# Removes all GLO entries not marked with C in \zstat
at GLO "(?s)\\p\s\\k\s[^\\]+\\k\*(?:(?!\\zstat\s[C]).)+?(?=\\p\s\\k\s[^\\]+\\k\*)" > ""

# Removes all GLO entries that are not at P:P1 or P:P2
# In other words, this prints only the priority numbers listed in []
# at GLO "(?s)\\p\s\\k\s[^\\]+\\k\*(?:(?!P:P[1]).)+?(?=\\p\s\\k\s[^\\]+\\k\*)" > ""

# Don't print verse numbers in glossary
at GLO "\\v \d+[\s\r\n]+" > ""

# Don't print status info in GLO
at GLO "\\zstat .+?[\r\n]*" > ""

# Don't print English gloss in GLO  - \zge ... \zge*
at GLO in "\\hp .+?[\r\n]+": "\\zge .+?\\zge\*;?" > ""

# Don't print priority info in GLO  - P:P2
at GLO in "\\hp .+?[\r\n]+": "P:.+?;" > ""

# Don't print category info in GLO  - CAT:ME
at GLO in "\\hp .+?[\r\n]+": "CAT:.+?;" > ""

# Don't print Greek words in GLO - G:
# at GLO in "\\hp .+?[\r\n]+": "G:.+?;" > ""

# Don't print Hebrew words in GLO - H:
# at GLO in "\\hp .+?[\r\n]+": "H:.+?;" > ""

# Wrap Greek words in wg
at GLO in "\\hp .+?[\r\n]+": "G:(.+?)([;\r\n\\])" > "\\wg \1\\wg*\2"

# Wrap Hebrew words in wg
at GLO in "\\hp .+?[\r\n]+": "H:(.+?)([;\r\n\\])" > "\\wh \1\\wh*\2"

# Finally discard the temp hp marker itself
at GLO "\\hp " > ""
at GLO "\\(zstat|hp)[\s\r\n]+" > ""

# This replaces the \p with \li only in the GLO book.
at GLO "\\p \\k " > "\\li \\k "
at GLO "\\p " > "\\pi "

 # Get rid of the references within captions of illustrations
# at GLOREF ".*" > ""

# This replaces the \p with \li only in the GLO book. 
at GLO "\\p \\k " > "\\li \\k "
at GLO "\\p " > "\\pi "


#  CONVERT ANGLED QUOTES TO ENGLISH (CURLY) QUOTES
#      -------------------------------------------------------------------------------
   '<< *<'               >    '\u201c \u2018' # Use nested open double and single curly quotes
                                                   #   (non-breaking thin space between)
   '<<'                  >    '\u201c'             # Use double open curly quotes
   '<'                   >    '\u2018'             # Use single open curly quotes 
   '> *>>'               >    '\u2019 \u201d' # Use nested close double and single curly quotes
                                                   #   (non-breaking thin space between)
   '>>'                  >    '\u201d'             # Use double close curly quotes
   '>'                   >    '\u2019'             # Use single close curly quotes

# CHOOSE HOW TO PROCESS LONG FOOTNOTES
# Keep footnotes marked with caller "@"
"\\f +@" > "\\f +"
# Remove footnotes marked with caller '@'
# "\\f +@.*?\\f\*" > "" 

#      CONVERT STANDARD ASCII QUOTES TO ENGLISH (CURLY) QUOTES
#      -------------------------------------------------------------------------------
#   "``"                  >    '\u201c'          # use double open  curly quotes
#   "`"                   >    '\u2018'          # use single open  curly quotes
#   '"'                   >    '\u201d'          # use double close curly quotes
#   "''"                  >    '\u201d'          # use double close curly quotes
#   "'"                   >    '\u2019'          # use double close curly quotes

#     REPLACE NORMAL SPACES BETWEEN NESTED QUOTES WITH THIN SPACES
#     OR INSERT A THIN SPACE IF THE NORMAL SPACE IS MISSING
#     -------------------------------------------------------------------------------
#   '(\u201c) *(\u2018)'  >    '\1\u2009\2'     # Place non-breaking thin space between
#  '(\u2018) *(\u201c)'  >    '\1\u2009\2'     #   nested open quote marks
#   '(\u2019) *(\u201d)'  >    '\1\u2009\2'     # Place non-breaking thin space between
#  '(\u201d) *(\u2019)'  >    '\1\u2009\2'     #   nested close quote marks
